---
title: "Get Customer"
description: "Retrieve a customer by ID."
api: "GET https://api.bipa.app/v1/customers/{customer_id}"
---

## Request

Retrieve details for a specific customer.

### Path parameters

<ParamField path="customer_id" type="string" required>
  The unique identifier of the customer (e.g., `cus_a1b2c3d4e5f6`)
</ParamField>

## Response

Returns the full customer object.

<ResponseField name="id" type="string">
  Unique customer identifier
</ResponseField>

<ResponseField name="type" type="string">
  `individual` or `business`
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `active`, `suspended`, `rejected`, `inactive`
</ResponseField>

<ResponseField name="document" type="object">
  <Expandable title="document">
    <ResponseField name="type" type="string">
      `cpf` or `cnpj`
    </ResponseField>
    <ResponseField name="number" type="string">
      Partially masked document number
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="name" type="string">
  Customer name (individuals) or trade name (businesses)
</ResponseField>

<ResponseField name="legal_name" type="string">
  Registered company name (business customers only)
</ResponseField>

<ResponseField name="email" type="string">
  Contact email
</ResponseField>

<ResponseField name="phone" type="string">
  Phone number
</ResponseField>

<ResponseField name="address" type="object">
  Full address object
</ResponseField>

<ResponseField name="metadata" type="object">
  Custom key-value data
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 creation timestamp
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 last update timestamp
</ResponseField>

<RequestExample>
```bash cURL
curl https://api.bipa.app/v1/customers/cus_a1b2c3d4e5f6 \
  -H "Authorization: Bearer sk_live_..."
```

```python Python
import requests

response = requests.get(
    "https://api.bipa.app/v1/customers/cus_a1b2c3d4e5f6",
    headers={"Authorization": "Bearer sk_live_..."}
)

customer = response.json()
```

```javascript Node.js
const response = await fetch(
  "https://api.bipa.app/v1/customers/cus_a1b2c3d4e5f6",
  {
    headers: { "Authorization": "Bearer sk_live_..." }
  }
);

const customer = await response.json();
```
</RequestExample>
