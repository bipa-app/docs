## Request

Create a customer representing a Brazilian company.

### Body parameters


<ParamField body="type" type="string" required>
  Must be `business`
</ParamField>

<ParamField body="document" type="string" required>
  CNPJ number (14 digits, numbers only).
</ParamField>

<ParamField body="legal_name" type="string" required>
  Registered company name (razão social)
</ParamField>

<ParamField body="trade_name" type="string">
  Doing business as / brand name (nome fantasia)
</ParamField>

<ParamField body="email" type="string" required>
  Company email address
</ParamField>

<ParamField body="phone" type="string" required>
  Phone number in E.164 format
</ParamField>

    </ParamField>
    <ParamField body="city" type="string" required>
      City name

    <ParamField body="state" type="string" required>
      Two-letter state code
    </ParamField>
    <ParamField body="postal_code" type="string" required>
      CEP (8 digits, numbers only)
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique customer identifier
</ResponseField>

<ResponseField name="status" type="string">
  Customer status. Possible values: `pending`, `active`, `rejected`
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp
</ResponseField>
