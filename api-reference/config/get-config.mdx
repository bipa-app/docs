## Request

Retrieve the current configuration for all supported assets and networks. This endpoint returns everything needed to build your integration: supported assets, networks, fees, limits, and decimal precision.

<Note>
  This endpoint does not require authentication and can be cached for up to 5 minutes.
</Note>

## Response

<ResponseField name="assets" type="object">
  Map of asset symbols to their configurations
  <Expandable title="Asset object">
    <ResponseField name="name" type="string">
      Human-readable asset name

    </ResponseField>
    <ResponseField name="decimals" type="integer">
      Number of decimal places for the smallest unit
    </ResponseField>
    <ResponseField name="networks" type="object">
      Map of network identifiers to their configurations
      <Expandable title="Network object">

          Human-readable network name
        </ResponseField>
        <ResponseField name="enabled" type="boolean">
          Whether the network is currently operational
        </ResponseField>
        <ResponseField name="deposit_enabled" type="boolean">
          Whether inbound transfers are enabled
        </ResponseField>
        <ResponseField name="withdrawal_enabled" type="boolean">
          Whether outbound transfers are enabled
        </ResponseField>
        <ResponseField name="withdrawal_fee" type="string">
          Fixed withdrawal fee in smallest unit (if `fee_type` is `fixed`)
        </ResponseField>
        <ResponseField name="withdrawal_fee_min" type="string">
          Minimum withdrawal fee in smallest unit (if `fee_type` is `dynamic`)
        </ResponseField>
        <ResponseField name="withdrawal_fee_max" type="string">
          Maximum withdrawal fee in smallest unit (if `fee_type` is `dynamic`)
        </ResponseField>
        <ResponseField name="fee_type" type="string">
          Fee type: `fixed` or `dynamic`
        </ResponseField>
        <ResponseField name="min_deposit" type="string">
          Minimum deposit amount in smallest unit
        </ResponseField>
        <ResponseField name="min_withdrawal" type="string">
          Minimum withdrawal amount in smallest unit
        </ResponseField>
        <ResponseField name="max_withdrawal" type="string">
          Maximum withdrawal amount per transaction in smallest unit
        </ResponseField>
        <ResponseField name="confirmations_required" type="integer">
          Number of blockchain confirmations required
        </ResponseField>
        <ResponseField name="estimated_deposit_minutes" type="integer">
          Estimated time for deposits to confirm
        </ResponseField>
        <ResponseField name="estimated_withdrawal_minutes" type="integer">
          Estimated time for withdrawals to complete
        </ResponseField>
        <ResponseField name="address_regex" type="string">
          Regex pattern for validating addresses on this network
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the configuration was last updated
</ResponseField>

```bash cURL
curl https://api.bipa.tech/v1/config
```

```python Python
import requests

response = requests.get("https://api.bipa.tech/v1/config")
config = response.json()

# Get USDT decimals
usdt_decimals = config["assets"]["USDT"]["decimals"]
print(f"USDT has {usdt_decimals} decimal places")

# List all enabled networks for USDT
for network_id, network in config["assets"]["USDT"]["networks"].items():
    if network["enabled"]:
        print(f"  {network['name']}: fee = {network.get('withdrawal_fee', 'dynamic')}")
```

```javascript Node.js
const response = await fetch("https://api.bipa.tech/v1/config");
const config = await response.json();

// Get USDT decimals
const usdtDecimals = config.assets.USDT.decimals;
console.log(`USDT has ${usdtDecimals} decimal places`);

// List all enabled networks for USDT
for (const [networkId, network] of Object.entries(config.assets.USDT.networks)) {
  if (network.enabled) {
    console.log(`  ${network.name}: fee = ${network.withdrawal_fee ?? 'dynamic'}`);
  }
}
```

## Usage examples

### Building a network selector

Use the config to dynamically build UI dropdowns:

```typescript
async function getNetworkOptions(asset: string) {
  const config = await fetch("https://api.bipa.tech/v1/config").then(r => r.json());

  const assetConfig = config.assets[asset];
  if (!assetConfig) return [];

  return Object.entries(assetConfig.networks)
    .filter(([_, network]) => network.enabled && network.withdrawal_enabled)
    .map(([id, network]) => ({
      value: id,
      label: network.name,
      fee: network.fee_type === "fixed"
        ? `${Number(network.withdrawal_fee) / 10 ** assetConfig.decimals} ${asset}`
        : `${Number(network.withdrawal_fee_min) / 10 ** assetConfig.decimals} - ${Number(network.withdrawal_fee_max) / 10 ** assetConfig.decimals} ${asset}`
    }));
}

// Returns:
// [
//   { value: "polygon", label: "Polygon", fee: "1 USDT" },
//   { value: "arbitrum", label: "Arbitrum One", fee: "1 USDT" },
//   { value: "ethereum", label: "Ethereum", fee: "3 - 25 USDT" },
//   ...
// ]
```

### Validating addresses client-side

```typescript
function validateAddress(asset: string, network: string, address: string, config: Config): boolean {
  const networkConfig = config.assets[asset]?.networks[network];
  if (!networkConfig) return false;

  const regex = new RegExp(networkConfig.address_regex);
  return regex.test(address);
}

// Example
const isValid = validateAddress("USDT", "polygon", "0x1234...", config);
```

### Converting amounts

```typescript
function toSmallestUnit(amount: string, asset: string, config: Config): string {
  const decimals = config.assets[asset]?.decimals ?? 6;
  const [whole, fraction = ""] = amount.split(".");
  const paddedFraction = fraction.padEnd(decimals, "0").slice(0, decimals);
  return BigInt(whole + paddedFraction).toString();
}

function toHumanReadable(smallestUnit: string, asset: string, config: Config): string {
  const decimals = config.assets[asset]?.decimals ?? 6;
  const value = BigInt(smallestUnit);
  const divisor = BigInt(10 ** decimals);
  const whole = value / divisor;
  const fraction = (value % divisor).toString().padStart(decimals, "0");
  return `${whole}.${fraction}`;
}
```

## Caching recommendations

<Tip>
  Cache this response for **5 minutes** in production. Configuration changes are infrequent and will be announced in advance.
</Tip>

```typescript
let configCache: { data: Config; expires: number } | null = null;

async function getConfig(): Promise<Config> {
  const now = Date.now();

  if (configCache && configCache.expires > now) {
    return configCache.data;
  }

  const response = await fetch("https://api.bipa.tech/v1/config");
  const data = await response.json();

  configCache = {
    data,
    expires: now + 5 * 60 * 1000 // 5 minutes
  };

  return data;
}
```
